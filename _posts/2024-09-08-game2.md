    <canvas id="gameCanvas" width="800" height="600"></canvas>

    <button id="navigateButton">Return to Site</button>

    <script>
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        
        // Game variables
        const playerSize = 30;
        const playerSpeed = 5;
        const playerSpeedFast = playerSpeed * 1.01; // Slightly faster for the non-it player
        const tagRadius = 40;
        const wallWidth = 20;
        const wallHeight = 100;
        
        let player1 = { x: 100, y: 100, color: 'red', dx: 0, dy: 0 };
        let player2 = { x: 600, y: 400, color: 'blue', dx: 0, dy: 0 };
        let walls = [];
        let keys = {};
        let score1 = 0;
        let score2 = 0;
        let turn = 'red'; // Current turn
        
        // Event listeners for keyboard input
        document.addEventListener('keydown', (e) => keys[e.key] = true);
        document.addEventListener('keyup', (e) => keys[e.key] = false);
        
        function drawPlayer(player) {
            ctx.beginPath();
            ctx.arc(player.x, player.y, playerSize / 2, 0, Math.PI * 2);
            ctx.fillStyle = player.color;
            ctx.fill();
            ctx.closePath();
        }
        
        function drawWalls() {
            ctx.fillStyle = '#333';
            walls.forEach(wall => {
                ctx.fillRect(wall.x, wall.y, wall.width, wall.height);
            });
        }
        
        function updatePlayerMovement(player, controls, speed) {
            player.dx = 0;
            player.dy = 0;
            
            if (keys[controls.left]) {
                player.dx = -speed;
            }
            if (keys[controls.right]) {
                player.dx = speed;
            }
            if (keys[controls.up]) {
                player.dy = -speed;
            }
            if (keys[controls.down]) {
                player.dy = speed;
            }
            
            // Check collisions with walls
            let nextX = player.x + player.dx;
            let nextY = player.y + player.dy;
            let collides = false;
            
            for (let wall of walls) {
                if (nextX + playerSize / 2 > wall.x && nextX - playerSize / 2 < wall.x + wall.width &&
                    nextY + playerSize / 2 > wall.y && nextY - playerSize / 2 < wall.y + wall.height) {
                    collides = true;
                    break;
                }
            }
            
            if (!collides) {
                player.x += player.dx;
                player.y += player.dy;
                
                // Wrap around the screen edges
                if (player.x < 0) player.x = canvas.width;
                if (player.x > canvas.width) player.x = 0;
                if (player.y < 0) player.y = canvas.height;
                if (player.y > canvas.height) player.y = 0;
            }
        }
        
        function checkTag() {
            const dist = Math.hypot(player1.x - player2.x, player1.y - player2.y);
            if (dist < tagRadius) {
                // Tag detected
                if (turn === 'red') {
                    score1++;
                    turn = 'blue';
                } else {
                    score2++;
                    turn = 'red';
                }
                resetPlayers();
            }
        }
        
        function resetPlayers() {
            do {
                player1.x = 100;
                player1.y = 100;
            } while (isPlayerOnWall(player1));
        
            do {
                player2.x = 600;
                player2.y = 400;
            } while (isPlayerOnWall(player2));
        }
        
        function isPlayerOnWall(player) {
            return walls.some(wall =>
                player.x + playerSize / 2 > wall.x && player.x - playerSize / 2 < wall.x + wall.width &&
                player.y + playerSize / 2 > wall.y && player.y - playerSize / 2 < wall.y + wall.height
            );
        }
        
        function drawScore() {
            ctx.font = '24px Arial';
            ctx.fillStyle = '#000';
            ctx.fillText(`Player 1 Score: ${score1}`, 100, 30);
            ctx.fillText(`Player 2 Score: ${score2}`, canvas.width - 100, 30);
            ctx.fillText(`Turn: ${turn}`, canvas.width / 2, 30);
        }
        
        function spawnWalls() {
            walls = [];
            for (let i = 0; i < 5; i++) {
                let newWall;
                do {
                    const x = Math.random() * (canvas.width - wallWidth);
                    const y = Math.random() * (canvas.height - wallHeight);
                    newWall = { x, y, width: wallWidth, height: wallHeight };
                } while (walls.some(wall => 
                    newWall.x < wall.x + wall.width &&
                    newWall.x + newWall.width > wall.x &&
                    newWall.y < wall.y + wall.height &&
                    newWall.y + newWall.height > wall.y
                ));
                walls.push(newWall);
            }
        }
        
        function drawGameStartMessage() {
            ctx.font = '30px Arial';
            ctx.fillStyle = '#000';
            ctx.textAlign = 'center';
            ctx.fillText(`Player ${turn.toUpperCase()} is the tagger first`, canvas.width / 2, canvas.height / 2);
        }
        
        function gameLoop() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            updatePlayerMovement(player1, { up: 'w', left: 'a', down: 's', right: 'd' }, turn === 'red' ? playerSpeed : playerSpeedFast);
            updatePlayerMovement(player2, { up: 'ArrowUp', left: 'ArrowLeft', down: 'ArrowDown', right: 'ArrowRight' }, turn === 'blue' ? playerSpeed : playerSpeedFast);
            
            drawWalls();
            drawPlayer(player1);
            drawPlayer(player2);
            drawScore();
            
            checkTag();
            
            requestAnimationFrame(gameLoop);
        }
        
        // Initialize game
        function startGame() {
            spawnWalls();
            drawGameStartMessage();
            setTimeout(() => {
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                gameLoop(); // Start the game loop after a short delay
            }, 2000); // 2-second delay
        }
        
        startGame();
        
            document.getElementById('navigateButton').addEventListener('click', () => {
                window.location.href = '/DavidCSA2025/blogs/';
            });
    </script>
